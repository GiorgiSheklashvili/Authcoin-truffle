        bytes32 out;

    function bytesToString(bytes src) pure internal returns (string) {
        uint l = src.length;
        bytes memory s = new bytes(l*2);
 
        for (uint i = 0; i < l; i++) {
            byte b = byte(src[i]);
            byte hi = byte(uint8(b) / 16);
            byte lo = byte(uint8(b) - 16 * uint8(hi));
            s[i*2] = char(hi);
            s[i*2+1] = char(lo);
        }
 
        return string(s);
    }
